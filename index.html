How to Use

<!-- Basic Usage -->
<button class="ui-btn ui-btn-primary">Primary Button</button>

<!-- With Size Variants -->
<button class="ui-btn ui-btn-secondary ui-btn-sm">Small Button</button>
<button class="ui-btn ui-btn-outline ui-btn-lg">Large Button</button>

<!-- JavaScript Factory Function -->
<script>
const button = createButton({
  text: 'Dynamic Button',
  variant: 'primary',
  size: 'default',
  onClick: () => console.log('Clicked!')
});
document.body.appendChild(button);
</script>
